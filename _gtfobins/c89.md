---
functions:
  file-read:
    - code: |
        LFILE=file_to_read
        c89 -x c -E "$LFILE"
  file-write:  # XXX this should be file-delete
    - code: |
        LFILE=file_to_delete
        c89 -xc /dev/null -o $LFILE
  shell:
    - code: c89 -wrapper /bin/sh,-s .
  sudo:
    - code: sudo c89 -wrapper /bin/sh,-s .
---
